    AI系统设计与架构

    概述

    AI系统设计是AI应用开发面试中的核心环节，考察候选人对从用户请求到模型推理的完整链路理解。本章涵盖AI应用整体架构、RAG系统、AI Agent、高并发设计、监控可观测性等主题，帮助候选人系统掌握AI系统设计的架构方法论和实践经验。

    知识点详解

    一、AI应用整体架构

    一个成熟的AI应用通常包含以下层次：

    [用户层]  Web/App/API客户端
    |
[网关层]  API Gateway、LB、Rate Limiter、Auth
    |
[编排层]  业务逻辑、Prompt管理、会话管理、Tool调度
    |
[推理层]  Model Router、推理引擎、模型Cache
    |
[存储层]  Vector DB、对象存储、Redis、PostgreSQL
    |
[监控层]  Metrics、Tracing、Logging、Cost分析

    1. 网关层（Gateway Layer）

    职责：

    • 身份认证与授权（API Key、JWT、OAuth）
    • 速率限制（Rate Limiting）
    • 请求路由与负载均衡
    • DDoS防护
    • 请求/响应日志

from fastapi import FastAPI, HTTPException, Depends
from fastapi.middleware.cors import CORSMiddleware
import time

app = FastAPI()

# 速率限制中间件
class RateLimiter:
    def __init__(self, max_requests: int = 100, window: int = 60):
        self.max_requests = max_requests
        self.window = window
        self.requests = {}

    async def __call__(self, request):
        client_ip = request.client.host
        now = time.time()
        if client_ip not in self.requests:
            self.requests[client_ip] = []
        # 清理过期记录
        self.requests[client_ip] = [
            t for t in self.requests[client_ip] if now - t < self.window
        ]
        if len(self.requests[client_ip]) >= self.max_requests:
            raise HTTPException(status_code=429, detail="Too Many Requests")
        self.requests[client_ip].append(now)

    # Prompt编排示例
class PromptOrchestrator:
    def __init__(self, system_template: str):
        self.system_template = system_template

    def build_prompt(self, user_input: str, history: list, context: dict = None):
        messages = [
            {"role": "system", "content": self.system_template.format(**context or {})},
        ]
        messages.extend(history)
        messages.append({"role": "user", "content": user_input})
        return messages

    def build_rag_prompt(self, query: str, documents: list, history: list):
        context = "\n\n".join([doc["content"] for doc in documents])
        rag_prompt = f"""基于以下上下文回答问题：

上下文：
{context}

用户问题：{query}"""
        return self.build_prompt(rag_prompt, history)

    # 模型路由器示例
class ModelRouter:
    def __init__(self):
        self.models = {
            "fast": {"endpoint": "http://vllm:8000", "model": "qwen2.5:7b", "cost": 0.5},
            "powerful": {"endpoint": "http://vllm:8001", "model": "llama3.1:70b", "cost": 2.0},
            "cheap": {"endpoint": "http://tgi:8080", "model": "qwen2.5:1.5b", "cost": 0.1},
        }

    def route(self, task_type: str, complexity: str):
        if task_type == "chat":
            if complexity == "simple":
                return self.models["fast"]
            return self.models["powerful"]
        elif task_type == "summarization":
            return self.models["fast"]
        elif task_type == "coding":
            return self.models["powerful"]
        return self.models["cheap"]

    def fallback(self, primary: str):
        # 主模型失败时切换到低成本模型
        return self.models["fast"]

    5. 监控层（Monitoring Layer）

    关键指标：

    • 延迟：P50/P95/P99 TTFT（Time to First Token）、TPOT（Time per Output Token）

    • 成本：每Token成本、每日总成本、模型级成本分布
    • 质量：用户满意度、回答准确率、幻觉率

# Rerank（重排序）
from sentence_transformers import CrossEncoder
reranker = CrossEncoder("BAAI/bge-reranker-v2-m3")

def hybrid_search_with_rerank(query: str, k: int = 3):
    results = ensemble_retriever.invoke(query)  # 先召回到20个
    pairs = [[query, doc.page_content] for doc in results]
    scores = reranker.predict(pairs)
    ranked = sorted(zip(results, scores), key=lambda x: x[1], reverse=True)
    return [doc for doc, score in ranked[:k]]

from openai import OpenAI

client = OpenAI(base_url="http://vllm:8000/v1")

def rag_generate(query: str):
    # 1. 检索
    docs = hybrid_search_with_rerank(query, k=5)

    # 2. 构建上下文
    context = "\n\n".join([
        f"文档{i+1}：{doc.page_content}"
        for i, doc in enumerate(docs)
    ])

    # 3. 生成
    response = client.chat.completions.create(
        model="llama3.1-70b",
        messages=[
            {"role": "system", "content": "基于以下上下文，用中文回答问题。如果不确定，请说明。"},
            {"role": "user", "content": f"上下文：\n{context}\n\n问题：{query}"}
        ],
        temperature=0.3,
        max_tokens=1024
    )
    return response.choices[0].message.content, docs

    import hashlib
import redis
import json

r = redis.Redis(host="localhost", port=6379, db=0)

class SemanticCache:
    """语义缓存：基于Embedding相似度判断缓存命中"""

    def __init__(self, threshold: float = 0.95):
        self.threshold = threshold
        self.cache_key_prefix = "semantic_cache:"

    def get(self, query: str) -> str | None:
        query_embedding = get_embedding(query)
        # 查找相似缓存
        cursor = 0
        while True:
            cursor, keys = r.scan(cursor, match=f"{self.cache_key_prefix}*")
            for key in keys:
                cached = json.loads(r.get(key))
                cached_emb = cached["embedding"]
                sim = cosine_similarity(query_embedding, cached_emb)
                if sim > self.threshold:
                    return cached["response"]
            if cursor == 0:
                break
        return None

    def set(self, query: str, response: str):
        key = f"{self.cache_key_prefix}{hashlib.md5(query.encode()).hexdigest()}"
        r.setex(key, 3600, json.dumps({
            "query": query,
            "embedding": get_embedding(query),
            "response": response
        }))

def cosine_similarity(a, b):
    import numpy as np
    a, b = np.array(a), np.array(b)
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

    from typing import Any
import json

class ToolRegistry:
    """工具注册与调度中心"""

    def __init__(self):
        self._tools = {}

    def register(self, tool: Any):
        self._tools[tool.name] = tool

    def get_tool_definitions(self) -> list:
        return [
            {
                "type": "function",
                "function": {
                    "name": tool.name,
                    "description": tool.description,
                    "parameters": tool.parameters
                }
            }
            for tool in self._tools.values()
        ]

    async def execute(self, tool_name: str, arguments: dict) -> str:
        if tool_name not in self._tools:
            return f"错误：工具 {tool_name} 不存在"
        tool = self._tools[tool_name]
        try:
            result = await tool.execute(**arguments)
            return json.dumps(result, ensure_ascii=False)
        except Exception as e:
            return f"工具执行错误：{str(e)}"

    from datetime import datetime
from typing import List, Dict
import uuid

class SessionManager:
    def __init__(self, max_history: int = 20, ttl: int = 3600):
        self.sessions: Dict[str, List[Dict]] = {}
        self.max_history = max_history
        self.ttl = ttl

    def create_session(self) -> str:
        session_id = str(uuid.uuid4())
        self.sessions[session_id] = []
        return session_id

    def add_message(self, session_id: str, role: str, content: str):
        if session_id not in self.sessions:
            self.sessions[session_id] = []
        self.sessions[session_id].append({
            "role": role,
            "content": content,
            "timestamp": datetime.now().isoformat()
        })
        # 限制历史长度
        if len(self.sessions[session_id]) > self.max_history:
            self.sessions[session_id] = self.sessions[session_id][-self.max_history:]

    def get_context(self, session_id: str, max_tokens: int = 4096) -> List[Dict]:
        messages = self.sessions.get(session_id, [])
        # 可以按token长度截断
        return [{"role": m["role"], "content": m["content"]} for m in messages]

    import asyncio
from tenacity import retry, stop_after_attempt, wait_exponential

class AgentTaskQueue:
    """Agent异步任务队列"""

    def __init__(self, max_retries: int = 3):
        self.task_queue = asyncio.Queue()
        self.max_retries = max_retries

    async def enqueue(self, task: dict):
        await self.task_queue.put(task)

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=1, max=10),
        reraise=True
    )
    async def process_task(self, task: dict):
        """带指数退避重试的任务处理"""
        tool_name = task["tool_name"]
        arguments = task["arguments"]
        result = await tool_registry.execute(tool_name, arguments)
        if "错误" in result:
            raise RuntimeError(result)
        return result

    async def worker(self):
        while True:
            task = await self.task_queue.get()
            try:
                result = await self.process_task(task)
                await task["callback"](result)
            except Exception as e:
                await task["error_callback"](str(e))
            finally:
                self.task_queue.task_done()

    import asyncio
from collections import defaultdict

class RequestBatcher:
    """动态请求合并器"""

    def __init__(self, max_batch_size: int = 8, max_wait_ms: int = 50):
        self.max_batch_size = max_batch_size
        self.max_wait_ms = max_wait_ms
        self.queues = defaultdict(list)
        self._locks = defaultdict(asyncio.Lock)

    async def submit(self, prompt: str, model: str) -> str:
        future = asyncio.get_event_loop().create_future()
        async with self._locks[model]:
            self.queues[model].append((prompt, future))
            if len(self.queues[model]) >= self.max_batch_size:
                asyncio.create_task(self._process_batch(model))
            else:
                # 启动延迟批处理
                asyncio.create_task(self._delayed_process(model))
        return await future

    async def _delayed_process(self, model: str):
        await asyncio.sleep(self.max_wait_ms / 1000)
        async with self._locks[model]:
            if self.queues[model]:
                asyncio.create_task(self._process_batch(model))

    async def _process_batch(self, model: str):
        batch = self.queues[model][:self.max_batch_size]
        self.queues[model] = self.queues[model][self.max_batch_size:]
        # 批量调用模型
        prompts = [item[0] for item in batch]
        results = await batch_inference(model, prompts)
        for (_, future), result in zip(batch, results):
            future.set_result(result)

    import time
from collections import defaultdict

class SlidingWindowRateLimiter:
    """滑动窗口速率限制器"""

    def __init__(self, max_requests: int = 60, window_seconds: int = 60):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.requests = defaultdict(list)

    def is_allowed(self, key: str = "default") -> bool:
        now = time.time()
        window_start = now - self.window_seconds
        # 清理过期请求
        self.requests[key] = [
            t for t in self.requests[key] if t > window_start
        ]
        if len(self.requests[key]) >= self.max_requests:
            return False
        self.requests[key].append(now)
        return True

    def get_wait_time(self, key: str = "default") -> float:
        if self.is_allowed(key):
            return 0
        if not self.requests[key]:
            return 0
        # 窗口中最旧的请求过期时间
        oldest = min(self.requests[key])
        return max(0, oldest + self.window_seconds - time.time())

    from enum import Enum
import asyncio

class CircuitState(Enum):
    CLOSED = "closed"       # 正常运行
    OPEN = "open"           # 熔断
    HALF_OPEN = "half_open" # 半开（尝试恢复）

class CircuitBreaker:
    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 30):
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.last_failure_time = 0

    async def call(self, func, fallback_func=None, *args, **kwargs):
        if self.state == CircuitState.OPEN:
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
            else:
                if fallback_func:
                    return await fallback_func(*args, **kwargs)
                raise Exception("Circuit breaker is OPEN")

        try:
            result = await func(*args, **kwargs)
            if self.state == CircuitState.HALF_OPEN:
                self.state = CircuitState.CLOSED
                self.failure_count = 0
            return result
        except Exception as e:
            self.failure_count += 1
            self.last_failure_time = time.time()
            if self.failure_count >= self.failure_threshold:
                self.state = CircuitState.OPEN
            if fallback_func:
                return await fallback_func(*args, **kwargs)
            raise e

    # 完整的数据ETL Pipeline示例
import asyncio
from typing import AsyncIterator

class DataPipeline:
    def __init__(self):
        self.steps = []

    def add_step(self, name: str, processor):
        self.steps.append((name, processor))
        return self

    async def process(self, data: AsyncIterator):
        async for item in data:
            for name, processor in self.steps:
                item = await processor(item)
                if item is None:
                    break
            if item is not None:
                yield item

# 使用示例

    import time
import random
from dataclasses import dataclass
from typing import Optional

@dataclass
class ModelConfig:
    name: str
    endpoint: str
    cost_per_1k_tokens: float
    max_tokens: int
    avg_latency_ms: float
    capabilities: list

class ModelRouter:
    def __init__(self):
        self.models = {
            "gpt-4o": ModelConfig(
                name="gpt-4o", endpoint="https://api.openai.com/v1",
                cost_per_1k_tokens=0.03, max_tokens=16384,
                avg_latency_ms=2000, capabilities=["chat", "vision", "coding", "complex"]
            ),
            "claude-3.5": ModelConfig(
                name="claude-3.5-sonnet", endpoint="https://api.anthropic.com/v1",
                cost_per_1k_tokens=0.02, max_tokens=8192,
                avg_latency_ms=1800, capabilities=["chat", "coding", "complex"]
            ),
            "llama3.1-70b": ModelConfig(
                name="llama3.1-70b", endpoint="http://vllm:8000/v1",
                cost_per_1k_tokens=0.005, max_tokens=8192,
                avg_latency_ms=2500, capabilities=["chat", "coding"]
            ),
            "qwen2.5-7b": ModelConfig(
                name="qwen2.5-7b", endpoint="http://vllm:8001/v1",
                cost_per_1k_tokens=0.001, max_tokens=4096,
                avg_latency_ms=500, capabilities=["chat", "simple"]
            ),
        }

    def route(self, task: dict) -> ModelConfig:
        """根据任务特征路由到最合适的模型"""
        complexity = task.get("complexity", "simple")
        latency_sla = task.get("latency_sla_ms", float("inf"))
        requires_vision = task.get("requires_vision", False)
        requires_coding = task.get("requires_coding", False)
        max_cost = task.get("max_cost", float("inf"))

        # 按成本排序，优先选低成本模型
        candidates = sorted(self.models.values(), key=lambda m: m.cost_per_1k_tokens)

        for model in candidates:
            if model.avg_latency_ms > latency_sla:
                continue
            if requires_vision and "vision" not in model.capabilities:
                continue
            if requires_coding and "coding" not in model.capabilities:
                continue
            if complexity == "complex" and "complex" not in model.capabilities:
                continue
            if model.cost_per_1k_tokens > max_cost:
                continue
            return model

        # Fallback到最便宜的模型
        return self.models["qwen2.5-7b"]

    def fallback(self, primary_model: str, error: Exception) -> ModelConfig:
        """主模型失败时切换到备用模型"""
        print(f"Model {primary_model} failed: {error}. Falling back...")
        # 按成本从低到高尝试
        for name, model in sorted(self.models.items(),
                                   key=lambda x: x[1].cost_per_1k_tokens):
            if name != primary_model:
                return model
        return self.models["qwen2.5-7b"]

    def a_b_test(self, task: dict, traffic_ratio: float = 0.1) -> tuple:
        """A/B测试：以指定比例路由到实验模型"""
        if random.random() < traffic_ratio:
            return "experimental", self.models["llama3.1-70b"]
        return "control", self.models["gpt-4o"]

    import re

class ContentSafety:
    def __init__(self):
        self.blocked_patterns = [
            r"(?i)暴力|色情|赌博|毒品|枪械|恐怖",
            r"(?i)hate|violence|porn|gambling|drugs",
        ]
        self.pii_patterns = [
            r"\b\d{18}\b",        # 身份证号
            r"\b1[3-9]\d{9}\b",   # 手机号
            r"\b\d{17}[\dXx]\b",  # 另一种身份证
        ]

    def check_input(self, text: str) -> dict:
        """检查用户输入是否安全"""
        for pattern in self.blocked_patterns:
            if re.search(pattern, text):
                return {"safe": False, "reason": "内容包含敏感词", "matched": pattern}
        return {"safe": True}

    def filter_output(self, text: str) -> str:
        """过滤模型输出中的敏感信息"""
        for pattern in self.pii_patterns:
            text = re.sub(pattern, "[REDACTED]", text)
        return text

    def audit_log(self, request: dict, response: dict):
        """输入输出审计日志"""
        log_entry = {
            "timestamp": time.time(),
            "user_id": request.get("user_id"),
            "input_hash": hashlib.sha256(request.get("input", "").encode()).hexdigest(),
            "output_hash": hashlib.sha256(response.get("output", "").encode()).hexdigest(),
            "model": request.get("model"),
            "tokens_used": response.get("tokens_used"),
            "safety_check": self.check_input(request.get("input", "")),
        }
        # 写入审计数据库
        audit_db.insert_one(log_entry)

    class RAGPermissionControl:
    """基于文档级别的RAG权限控制"""

    def __init__(self):
        # 用户 -> 可访问的文档标签
        self.user_permissions = {
            "user_a": ["public", "team_a"],
            "user_b": ["public", "team_b", "confidential"],
            "admin": ["public", "team_a", "team_b", "confidential", "secret"],
        }

    def filter_documents(self, user_id: str, documents: list) -> list:
        """根据用户权限过滤可检索的文档"""
        allowed_tags = self.user_permissions.get(user_id, ["public"])
        return [
            doc for doc in documents
            if doc.get("security_tag", "public") in allowed_tags
        ]

    def hybrid_search_with_permissions(self, user_id: str, query: str):
        # 1. 执行检索
        all_docs = vector_store.similarity_search(query, k=20)
        # 2. 按权限过滤
        allowed_docs = self.filter_documents(user_id, all_docs)
        # 3. 返回权限范围内的结果
        return allowed_docs[:5]

    class ModelCascade:
    """模型级联：先用小模型，复杂度高时升级到大模型"""

    def __init__(self):
        self.small_model = "qwen2.5-1.5b"   # 成本低、速度快
        self.medium_model = "qwen2.5-7b"     # 适中
        self.large_model = "llama3.1-70b"    # 能力强、成本高

    async def infer_with_cascade(self, prompt: str) -> str:
        result = await self._call_model(self.small_model, prompt)
        if self._is_confident(result):
            return result
        # 小模型不够自信，升级到中模型
        result = await self._call_model(self.medium_model, prompt)
        if self._is_confident(result):
            return result
        # 最后使用大模型
        return await self._call_model(self.large_model, prompt)

    def _is_confident(self, result: str) -> bool:
        # 检查模型是否"不确定"
        uncertainty_markers = ["不确定", "不清楚", "我不知道", "I'm not sure", "cannot"]
        return not any(marker in result.lower() for marker in uncertainty_markers)
