    import numpy as np

def scaled_dot_product_attention(Q, K, V, mask=None):
    """
    Scaled Dot-Product Attention 实现
    
    Args:
        Q: Query矩阵, shape (batch_size, seq_len, d_k)
        K: Key矩阵, shape (batch_size, seq_len, d_k)
        V: Value矩阵, shape (batch_size, seq_len, d_v)
        mask: 可选mask, shape (batch_size, seq_len, seq_len)
    
    Returns:
        output: Attention输出
        attention_weights: 注意力权重
    """
    d_k = Q.shape[-1]
    
    # 1. 计算QK^T / sqrt(d_k)
    scores = np.matmul(Q, K.transpose(0, 2, 1)) / np.sqrt(d_k)
    
    # 2. 应用mask（如果有）
    if mask is not None:
        scores = np.where(mask == 0, -1e9, scores)
    
    # 3. Softmax
    exp_scores = np.exp(scores - np.max(scores, axis=-1, keepdims=True))
    attention_weights = exp_scores / np.sum(exp_scores, axis=-1, keepdims=True)
    
    # 4. 加权求和
    output = np.matmul(attention_weights, V)
    
    return output, attention_weights


def multi_head_attention(Q, K, V, num_heads, d_model, mask=None):
    """
    多头注意力实现
    
    Args:
        Q, K, V: 输入矩阵
        num_heads: 注意力头数
        d_model: 模型维度
    """
    batch_size = Q.shape[0]
    d_k = d_model // num_heads
    
    # 1. 线性变换并分头
    def split_heads(x):
        x = x.reshape(batch_size, -1, num_heads, d_k)
        return x.transpose(0, 2, 1, 3)  # (batch, heads, seq, d_k)
    
    Q_heads = split_heads(Q)
    K_heads = split_heads(K)
    V_heads = split_heads(V)
    
    # 2. 对每个头计算attention
    output_heads, attn_weights = scaled_dot_product_attention(
        Q_heads, K_heads, V_heads, mask
    )
    
    # 3. 合并头
    output = output_heads.transpose(0, 2, 1, 3)  # (batch, seq, heads, d_k)
    output = output.reshape(batch_size, -1, d_model)
    
    return output

    import numpy as np

def top_k_sampling(logits, k=50, temperature=1.0):
    """
    Top-K采样实现
    
    Args:
        logits: 模型输出logits, shape (vocab_size,)
        k: 保留概率最高的k个token
        temperature: 温度参数（>1更随机，<1更确定）
    
    Returns:
        sampled_token: 采样得到的token id
    """
    # 1. 应用温度
    logits = logits / temperature
    
    # 2. 取Top-K
    top_k_values = -np.sort(-logits)[:k]
    top_k_indices = np.argsort(-logits)[:k]
    threshold = top_k_values[-1]
    
    # 3. 将小于阈值的logits设为负无穷
    filtered_logits = np.where(
        logits >= threshold, logits, -float('Inf')
    )
    
    # 4. Softmax转换为概率
    exp_logits = np.exp(filtered_logits - np.max(filtered_logits))
    probs = exp_logits / np.sum(exp_logits)
    
    # 5. 根据概率采样
    sampled_token = np.random.choice(len(probs), p=probs)
    
    return sampled_token


def top_p_sampling(logits, p=0.9, temperature=1.0):
    """
    Top-P (Nucleus) 采样实现
    """
    logits = logits / temperature
    sorted_indices = np.argsort(logits)[::-1]
    sorted_logits = logits[sorted_indices]
    
    # 计算累积概率
    exp_logits = np.exp(sorted_logits - np.max(sorted_logits))
    probs = exp_logits / np.sum(exp_logits)
    cum_probs = np.cumsum(probs)
    
    # 找到累积概率超过p的阈值
    threshold_idx = np.where(cum_probs > p)[0][0]
    threshold = sorted_logits[threshold_idx]
    
    # 过滤
    filtered_logits = np.where(logits >= threshold, logits, -float('Inf'))
    exp_logits = np.exp(filtered_logits - np.max(filtered_logits))
    probs = exp_logits / np.sum(exp_logits)
    
    return np.random.choice(len(probs), p=probs)

    import numpy as np

class KVCache:
    """KV Cache模拟实现"""
    
    def __init__(self, max_batch_size=1, max_seq_len=4096, num_layers=32,
                 num_heads=32, head_dim=128):
        self.max_batch_size = max_batch_size
        self.max_seq_len = max_seq_len
        self.num_layers = num_layers
        self.num_heads = num_heads
        self.head_dim = head_dim
        
        # 初始化KV Cache（预分配内存）
        self.k_cache = np.zeros(
            (num_layers, max_batch_size, num_heads, max_seq_len, head_dim),
            dtype=np.float16
        )
        self.v_cache = np.zeros_like(self.k_cache)
        self.current_seq_len = 0
    
    def update(self, layer_idx, k_new, v_new):
        """
        更新KV Cache（追加新的K、V）
        
        Args:
            layer_idx: 层索引
            k_new: 新的Key, shape (batch, heads, new_tokens, dim)
            v_new: 新的Value, shape (batch, heads, new_tokens, dim)
        """
        new_tokens = k_new.shape[-2]
        start = self.current_seq_len
        end = start + new_tokens
        
        self.k_cache[layer_idx, :, :, start:end, :] = k_new
        self.v_cache[layer_idx, :, :, start:end, :] = v_new
        
        self.current_seq_len = end
    
    def get(self, layer_idx):
        """获取当前缓存的K、V"""
        return (
            self.k_cache[layer_idx, :, :, :self.current_seq_len, :],
            self.v_cache[layer_idx, :, :, :self.current_seq_len, :]
        )
    
    def reset(self):
        """重置Cache（新序列时调用）"""
        self.current_seq_len = 0
    
    def memory_usage_gb(self):
        """计算KV Cache显存占用（GB）"""
        bytes_per_element = 2  # float16
        total_elements = self.k_cache.size + self.v_cache.size
        return total_elements * bytes_per_element / (1024 ** 3)

class AutoregressiveGenerator:
    def __init__(self, vocab_size=32000, d_model=4096, num_layers=32,
                 num_heads=32):
        self.vocab_size = vocab_size
        self.d_model = d_model
        self.num_layers = num_layers
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        self.kv_cache = KVCache(
            num_layers=num_layers,
            num_heads=num_heads,
            head_dim=self.head_dim
        )
    
    def generate(self, input_ids, max_new_tokens=100):
        """模拟带KV Cache的自回归生成"""
        generated = list(input_ids)
        
        # Prefill阶段：处理所有输入token
        for pos, token_id in enumerate(input_ids):
            # 模拟单token的KV计算
            k = np.random.randn(1, self.num_heads, 1, self.head_dim).astype(np.float16)
            v = np.random.randn(1, self.num_heads, 1, self.head_dim).astype(np.float16)
            
            # 更新KV Cache
            for layer in range(self.num_layers):
                self.kv_cache.update(layer, k, v)
        
        # Decode阶段：逐token生成
        for _ in range(max_new_tokens):
            # 只输入最后一个token
            last_token = generated[-1]
            
            # 计算当前token的K、V
            k_new = np.random.randn(1, self.num_heads, 1, self.head_dim).astype(np.float16)
            v_new = np.random.randn(1, self.num_heads, 1, self.head_dim).astype(np.float16)
            
            # 更新KV Cache（追加）
            for layer in range(self.num_layers):
                self.kv_cache.update(layer, k_new, v_new)
            
            # 模拟预测下一个token
            logits = np.random.randn(self.vocab_size)
            next_token = np.argmax(logits)
            generated.append(next_token)
            
            if next_token == 0:  # EOS token
                break
        
        return generated

from langchain_community.document_loaders import (

from langchain_core.documents import Document

class DocumentLoader:
    """多格式文档加载器"""
    
    SUPPORTED_EXTENSIONS = {
        ".pdf": PyPDFLoader,
        ".txt": TextLoader,
        ".md": TextLoader,
    }
    
    @classmethod
    def load_file(cls, file_path: str) -> List[Document]:
        ext = os.path.splitext(file_path)[1].lower()
        loader_class = cls.SUPPORTED_EXTENSIONS.get(ext)
        if not loader_class:
            raise ValueError(f"Unsupported file type: {ext}")
        
        loader = loader_class(file_path)
        return loader.load()
    
    @classmethod
    def load_directory(cls, dir_path: str) -> List[Document]:
        docs = []
        for ext, loader_class in cls.SUPPORTED_EXTENSIONS.items():
            loader = DirectoryLoader(
                dir_path,
                glob=f"**/*{ext}",
                loader_cls=loader_class,
                show_progress=True
            )
            docs.extend(loader.load())
        return docs
    
    @classmethod
    def load_webpage(cls, url: str) -> List[Document]:
        loader = WebBaseLoader(url)
        return loader.load()

from langchain_text_splitters import (
    RecursiveCharacterTextSplitter,
    TokenTextSplitter,
)

class TextChunker:
    """文本切分器"""
    
    @staticmethod
    def recursive_chunk(
        documents: List[Document],
        chunk_size: int = 500,
        chunk_overlap: int = 50,
    ) -> List[Document]:
        """递归字符切分（推荐）"""
        splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            separators=["\n\n", "\n", "。", ".", " ", ""],
            length_function=len,
        )
        return splitter.split_documents(documents)
    
    @staticmethod
    def token_chunk(
        documents: List[Document],
        chunk_size: int = 300,
        chunk_overlap: int = 30,
    ) -> List[Document]:
        """基于Token的切分"""
        splitter = TokenTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
        )
        return splitter.split_documents(documents)
    
    @staticmethod
    def semantic_chunk(
        documents: List[Document],
        max_chunk_size: int = 500,
        min_chunk_size: int = 100,
    ) -> List[Document]:
        """
        语义切分：按段落边界切分，尽量保持语义完整
        适合文档结构清晰的场景
        """
        chunks = []
        for doc in documents:
            paragraphs = doc.page_content.split("\n\n")
            current_chunk = []
            current_size = 0
            
            for para in paragraphs:
                para_size = len(para)
                if current_size + para_size > max_chunk_size and current_chunk:
                    chunks.append(Document(
                        page_content="\n\n".join(current_chunk),
                        metadata=doc.metadata
                    ))
                    current_chunk = [para]
                    current_size = para_size
                else:
                    current_chunk.append(para)
                    current_size += para_size
            
            if current_chunk:
                chunks.append(Document(
                    page_content="\n\n".join(current_chunk),
                    metadata=doc.metadata
                ))
        
        return chunks

from langchain_openai import OpenAIEmbeddings
from langchain_community.embeddings import HuggingFaceEmbeddings
from langchain_community.vectorstores import Chroma

class VectorStore:
    """向量存储管理"""
    
    def __init__(
        self,
        embedding_model: str = "BAAI/bge-small-zh-v1.5",
        persist_directory: str = "./chroma_db",
    ):
        self.embedding_model = embedding_model
        self.persist_directory = persist_directory
        
        # 初始化Embedding模型
        self.embeddings = HuggingFaceEmbeddings(
            model_name=embedding_model,
            model_kwargs={"device": "cpu"},
            encode_kwargs={"normalize_embeddings": True},
        )
        
        # 初始化或加载向量库
        self.vectorstore = None
        if os.path.exists(persist_directory):
            self.vectorstore = Chroma(
                persist_directory=persist_directory,
                embedding_function=self.embeddings,
            )
    
    def create_index(self, chunks: List[Document]):
        """创建向量索引"""
        # 为每个chunk添加唯一ID
        for i, chunk in enumerate(chunks):
            chunk.metadata["chunk_id"] = hashlib.md5(
                chunk.page_content.encode()
            ).hexdigest()[:12]
        
        self.vectorstore = Chroma.from_documents(
            documents=chunks,
            embedding=self.embeddings,
            persist_directory=self.persist_directory,
        )
        self.vectorstore.persist()
        print(f"Created index with {len(chunks)} chunks")
    
    def similarity_search(
        self, query: str, k: int = 5
    ) -> List[Document]:
        """向量相似度检索"""
        if not self.vectorstore:
            raise ValueError("Vector store not initialized")
        return self.vectorstore.similarity_search(query, k=k)
    
    def similarity_search_with_score(
        self, query: str, k: int = 5
    ) -> List[Tuple[Document, float]]:
        """带分数的相似度检索"""
        if not self.vectorstore:
            raise ValueError("Vector store not initialized")
        return self.vectorstore.similarity_search_with_score(query, k=k)

from openai import OpenAI
import numpy as np

class RAGSystem:
    """完整RAG系统"""
    
    def __init__(
        self,
        embedding_model: str = "BAAI/bge-small-zh-v1.5",
        llm_endpoint: str = "http://localhost:8000/v1",
        llm_model: str = "qwen2.5-7b",
        persist_directory: str = "./chroma_db",
    ):
        # 向量存储
        self.vector_store = VectorStore(
            embedding_model=embedding_model,
            persist_directory=persist_directory,
        )
        
        # LLM客户端
        self.llm_client = OpenAI(
            base_url=llm_endpoint,
            api_key="not-needed"
        )
        self.llm_model = llm_model
        
        # 检索设置
        self.top_k = 5
        self.score_threshold = 0.7
    
    def index_documents(self, docs: List[Document]):
        """索引文档"""
        chunks = TextChunker.recursive_chunk(docs)
        self.vector_store.create_index(chunks)
    
    def retrieve(self, query: str) -> List[Document]:
        """检索相关文档"""
        results = self.vector_store.similarity_search_with_score(
            query, k=self.top_k
        )
        # 按分数阈值过滤
        relevant_docs = [
            doc for doc, score in results
            if score >= self.score_threshold
        ] if self.score_threshold else [doc for doc, _ in results]
        return relevant_docs
    
    def generate(self, query: str, context_docs: List[Document]) -> str:
        """基于检索结果生成回答"""
        context = "\n\n".join([
            f"[文档{i+1}] {doc.page_content}"
            for i, doc in enumerate(context_docs)
        ])
        
        system_prompt = """你是一个专业的AI助手，请基于以下提供的上下文准确回答用户问题。
如果上下文中没有足够信息，请明确说明"根据提供的资料无法回答此问题"。
请用中文回答，并引用相关文档编号。"""
        
        user_prompt = f"""上下文：
{context}

用户问题：{query}

请基于上下文回答问题："""
        
        response = self.llm_client.chat.completions.create(
            model=self.llm_model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt},
            ],
            temperature=0.3,
            max_tokens=1024,
        )
        
        return response.choices[0].message.content
    
    def query(self, query: str) -> dict:
        """完整RAG查询流程"""
        # 1. 检索
        retrieved_docs = self.retrieve(query)
        
        if not retrieved_docs:
            return {
                "query": query,
                "answer": "未检索到相关文档。",
                "sources": [],
                "has_context": False,
            }
        
        # 2. 生成
        answer = self.generate(query, retrieved_docs)
        
        # 3. 返回结果
        sources = [
            {
                "content": doc.page_content[:200] + "...",
                "source": doc.metadata.get("source", "unknown"),
            }
            for doc in retrieved_docs[:3]
        ]
        
        return {
            "query": query,
            "answer": answer,
            "sources": sources,
            "has_context": True,
        }

    """
ReAct (Reasoning + Acting) Agent 实现
思想：将推理步骤和工具调用交替进行
"""

import json
import re
from typing import Callable, Dict, List, Any
from openai import OpenAI


class Tool:
    """工具定义"""
    
    def __init__(self, name: str, description: str, func: Callable,
                 parameters: dict):
        self.name = name
        self.description = description
        self.func = func
        self.parameters = parameters
    
    def to_openai_tool(self) -> dict:
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.parameters,
            }
        }
    
    def execute(self, **kwargs) -> str:
        try:
            result = self.func(**kwargs)
            return str(result)
        except Exception as e:
            return f"Error: {str(e)}"


class ReActAgent:
    """
    ReAct Agent实现
    
    工作流程：
    1. 思考（Thought）：决定下一步做什么
    2. 行动（Action）：执行工具调用
    3. 观察（Observation）：获取工具执行结果
    4. 循环直到得到最终答案
    """
    
    def __init__(self, tools: List[Tool], model: str = "gpt-4",
                 max_steps: int = 10):
        self.tools = {tool.name: tool for tool in tools}
        self.tool_definitions = [t.to_openai_tool() for t in tools]
        self.model = model
        self.max_steps = max_steps
        self.client = OpenAI()
    
    def run(self, user_input: str) -> str:
        """运行ReAct Agent"""
        messages = [
            {"role": "system", "content": self._system_prompt()},
            {"role": "user", "content": user_input},
        ]
        
        steps = []
        
        for step in range(self.max_steps):
            print(f"\n=== Step {step + 1} ===")
            
            # 调用LLM
            response = self.client.chat.completions.create(
                model=self.model,
                messages=messages,
                tools=self.tool_definitions,
                tool_choice="auto",
                temperature=0.7,
            )
            
            assistant_message = response.choices[0].message
            
            # 检查是否有工具调用
            if not assistant_message.tool_calls:
                # 没有工具调用 -> 最终答案
                final_answer = assistant_message.content
                print(f"Final answer: {final_answer}")
                return final_answer
            
            # 处理工具调用
            messages.append(assistant_message)
            
            for tool_call in assistant_message.tool_calls:
                tool_name = tool_call.function.name
                arguments = json.loads(tool_call.function.arguments)
                
                print(f"Thought: 需要调用 {tool_name}")
                print(f"Action: {tool_name}({arguments})")
                
                # 执行工具
                if tool_name in self.tools:
                    result = self.tools[tool_name].execute(**arguments)
                else:
                    result = f"Error: Tool {tool_name} not found"
                
                print(f"Observation: {result[:200]}")
                
                # 添加工具响应
                messages.append({
                    "role": "tool",
                    "tool_call_id": tool_call.id,
                    "content": result,
                })
                
                steps.append({
                    "step": step + 1,
                    "tool": tool_name,
                    "args": arguments,
                    "result": result[:200],
                })
        
        return "Max steps reached without final answer."
    
    def _system_prompt(self) -> str:
        tool_descriptions = "\n".join([
            f"- {tool.name}: {tool.description}"
            for tool in self.tools.values()
        ])
        return f"""你是一个使用工具的AI助手。请按照以下格式工作：

可用工具：
{tool_descriptions}

工作流程：
1. 分析用户问题，思考需要使用哪些工具
2. 调用合适的工具获取信息
3. 根据工具返回结果继续思考
4. 当获取足够信息后，给出最终答案

注意：
- 每次只调用一个工具
- 调用工具时使用精确的参数
- 如果工具返回错误，尝试其他方法
- 重要：当不再需要调用工具时，直接回复最终答案"""

    """
Function Calling 的回调实现
支持同步和异步回调、超时控制、错误处理
"""

import asyncio
import functools
import time
from typing import Any, Callable, Dict, Optional
from dataclasses import dataclass


@dataclass
class FunctionCall:
    """函数调用封装"""
    name: str
    arguments: Dict[str, Any]
    call_id: str
    timestamp: float


@dataclass
class FunctionResult:
    """函数调用结果"""
    call_id: str
    name: str
    result: Any
    success: bool
    error: Optional[str] = None
    duration_ms: float = 0.0


class FunctionCallingHandler:
    """
    函数调用处理器
    
    功能：
    - 注册和管理函数
    - 执行函数调用（同步/异步）
    - 超时控制
    - 错误处理和重试
    - 调用日志记录
    """
    
    def __init__(self, timeout: float = 30.0, max_retries: int = 2):
        self._functions: Dict[str, Callable] = {}
        self._definitions: list = []
        self.timeout = timeout
        self.max_retries = max_retries
        self.call_log: list = []
    
    def register(self, func: Callable = None, *, name: str = None,
                 description: str = None, parameters: dict = None):
        """
        注册函数（支持装饰器用法）
        
        @handler.register
        def my_func(...): ...
        
        @handler.register(name="custom_name", ...)
        def my_func(...): ...
        """
        if func is None:
            return functools.partial(
                self.register, name=name,
                description=description, parameters=parameters
            )
        
        func_name = name or func.__name__
        func_description = description or (func.__doc__ or "").strip()
        
        self._functions[func_name] = func
        self._definitions.append({
            "type": "function",
            "function": {
                "name": func_name,
                "description": func_description,
                "parameters": parameters or {
                    "type": "object",
                    "properties": {},
                    "required": [],
                },
            },
        })
        
        return func
    
    async def execute_async(self, call: FunctionCall) -> FunctionResult:
        """异步执行函数调用"""
        start_time = time.time()
        
        if call.name not in self._functions:
            return FunctionResult(
                call_id=call.call_id,
                name=call.name,
                result=None,
                success=False,
                error=f"Function '{call.name}' not registered",
                duration_ms=(time.time() - start_time) * 1000,
            )
        
        func = self._functions[call.name]
        
        for attempt in range(self.max_retries + 1):
            try:
                if asyncio.iscoroutinefunction(func):
                    result = await asyncio.wait_for(
                        func(**call.arguments),
                        timeout=self.timeout,
                    )
                else:
                    result = await asyncio.wait_for(
                        asyncio.to_thread(func, **call.arguments),
                        timeout=self.timeout,
                    )
                
                fr = FunctionResult(
                    call_id=call.call_id,
                    name=call.name,
                    result=result,
                    success=True,
                    duration_ms=(time.time() - start_time) * 1000,
                )
                
                self.call_log.append(fr)
                return fr
                
            except asyncio.TimeoutError:
                error = f"Function '{call.name}' timed out after {self.timeout}s"
                if attempt < self.max_retries:
                    print(f"{error}, retrying ({attempt + 1}/{self.max_retries})")
                    continue
                fr = FunctionResult(
                    call_id=call.call_id,
                    name=call.name,
                    result=None,
                    success=False,
                    error=error,
                    duration_ms=(time.time() - start_time) * 1000,
                )
                self.call_log.append(fr)
                return fr
                
            except Exception as e:
                error = f"Function '{call.name}' failed: {str(e)}"
                if attempt < self.max_retries:
                    print(f"{error}, retrying ({attempt + 1}/{self.max_retries})")
                    continue
                fr = FunctionResult(
                    call_id=call.call_id,
                    name=call.name,
                    result=None,
                    success=False,
                    error=error,
                    duration_ms=(time.time() - start_time) * 1000,
                )
                self.call_log.append(fr)
                return fr
    
    async def execute_all(self, calls: list) -> list:
        """批量并行执行多个函数调用"""
        tasks = [self.execute_async(call) for call in calls]
        return await asyncio.gather(*tasks)
    
    @property
    def definitions(self) -> list:
        """获取OpenAI格式的工具定义"""
        return self._definitions
    
    def get_call_stats(self) -> dict:
        """获取调用统计"""
        if not self.call_log:
            return {}
        
        total = len(self.call_log)
        success = sum(1 for c in self.call_log if c.success)
        avg_duration = sum(c.duration_ms for c in self.call_log) / total
        
        return {
            "total_calls": total,
            "success_rate": f"{success / total * 100:.1f}%",
            "avg_duration_ms": f"{avg_duration:.1f}",
        }

def build_compressed_context(history: list, max_tokens: int = 2000) -> str:
    """
    压缩历史对话：对过长的历史进行摘要压缩
    """
    import tiktoken
    
    encoder = tiktoken.get_encoding("cl100k_base")
    
    full_history = ""
    for user_msg, asst_msg in history:
        turn = f"用户: {user_msg}\n助手: {asst_msg}\n"
        full_history += turn
    
    tokens = encoder.encode(full_history)
    if len(tokens) <= max_tokens:
        return full_history
    
    # 需要压缩：保留最近的对话，对早期的进行摘要
    recent_turns = []
    recent_tokens = 0
    half_tokens = max_tokens // 2
    
    for user_msg, asst_msg in reversed(history):
        turn_text = f"用户: {user_msg}\n助手: {asst_msg}\n"
        turn_tokens = len(encoder.encode(turn_text))
        if recent_tokens + turn_tokens > half_tokens:
            break
        recent_turns.insert(0, (user_msg, asst_msg))
        recent_tokens += turn_tokens
    
    # 早期历史用摘要替代
    early_turns = history[:-len(recent_turns)] if len(recent_turns) < len(history) else []
    
    if early_turns:
        summary = f"[早期对话摘要：共{len(early_turns)}轮，主要内容涉及{early_turns[0][0][:50]}...]\n"
    else:
        summary = ""
    
    recent_text = "\n".join([
        f"用户: {u}\n助手: {a}" for u, a in recent_turns
    ])
    
    return summary + recent_text
