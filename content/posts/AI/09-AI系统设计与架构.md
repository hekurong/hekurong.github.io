### 一、AI应用整体架构

一个成熟的AI应用通常包含以下层次：

```
[用户层]  Web/App/API客户端
    |
[网关层]  API Gateway、LB、Rate Limiter、Auth
    |
[编排层]  业务逻辑、Prompt管理、会话管理、Tool调度
    |
[推理层]  Model Router、推理引擎、模型Cache
    |
[存储层]  Vector DB、对象存储、Redis、PostgreSQL
    |
[监控层]  Metrics、Tracing、Logging、Cost分析
```

```python
# 速率限制中间件
class RateLimiter:
    def __init__(self, max_requests: int = 100, window: int = 60):
        self.max_requests = max_requests
        self.window = window
        self.requests = {}

    async def __call__(self, request):
        client_ip = request.client.host
        now = time.time()
        if client_ip not in self.requests:
            self.requests[client_ip] = []
        # 清理过期记录
        self.requests[client_ip] = [
            t for t in self.requests[client_ip] if now - t < self.window
        ]
        if len(self.requests[client_ip]) >= self.max_requests:
            raise HTTPException(status_code=429, detail="Too Many Requests")
        self.requests[client_ip].append(now)
```

```python
class PromptOrchestrator:
    def __init__(self, system_template: str):
        self.system_template = system_template

    def build_prompt(self, user_input: str, history: list, context: dict = None):
        messages = [

用户问题：{query}"""
```

```python
        elif task_type == "coding":
            return self.models["powerful"]
```

#### 2. 检索服务

检索策略演进：

| 策略 | 原理 | 适用场景 |
|------|------|---------|
| 朴素向量检索 | Cosine Similarity | 通用场景 |
| 混合检索 | Dense + Sparse (BM25) | 关键词敏感场景 |
| 重排序（Rerank） | Cross-Encoder二次排序 | 高精度场景 |
| 多路召回 | 多维度分别检索后合并 | 复杂查询 |
| 父文档检索 | 检索子chunk返回父文档 | 需要完整上下文 |

```python

from langchain_community.vectorstores import FAISS

# 向量检索
vectorstore = FAISS.from_documents(chunks, embeddings)
vector_retriever = vectorstore.as_retriever(search_kwargs={"k": 5})

# BM25检索
bm25_retriever = BM25Retriever.from_documents(chunks)
bm25_retriever.k = 5

# 集成检索
ensemble_retriever = EnsembleRetriever(
    retrievers=[bm25_retriever, vector_retriever],
    weights=[0.3, 0.7]
)

# Rerank（重排序）
from sentence_transformers import CrossEncoder
reranker = CrossEncoder("BAAI/bge-reranker-v2-m3")

def hybrid_search_with_rerank(query: str, k: int = 3):
    results = ensemble_retriever.invoke(query)  # 先召回到20个
    pairs = [[query, doc.page_content] for doc in results]
    scores = reranker.predict(pairs)
    ranked = sorted(zip(results, scores), key=lambda x: x[1], reverse=True)
    return [doc for doc, score in ranked[:k]]
```

#### 4. 缓存策略

```python
import hashlib
import redis
import json

r = redis.Redis(host="localhost", port=6379, db=0)

class SemanticCache:
    """语义缓存：基于Embedding相似度判断缓存命中"""

    def __init__(self, threshold: float = 0.95):
        self.threshold = threshold
        self.cache_key_prefix = "semantic_cache:"

    def get(self, query: str) -> str | None:
        query_embedding = get_embedding(query)
        # 查找相似缓存
        cursor = 0
        while True:
            cursor, keys = r.scan(cursor, match=f"{self.cache_key_prefix}*")
            for key in keys:
                cached = json.loads(r.get(key))
                cached_emb = cached["embedding"]
                sim = cosine_similarity(query_embedding, cached_emb)
                if sim > self.threshold:
                    return cached["response"]
            if cursor == 0:
                break
        return None

    def set(self, query: str, response: str):
        key = f"{self.cache_key_prefix}{hashlib.md5(query.encode()).hexdigest()}"
        r.setex(key, 3600, json.dumps({
            "query": query,
            "embedding": get_embedding(query),
            "response": response
        }))

def cosine_similarity(a, b):
    import numpy as np
    a, b = np.array(a), np.array(b)
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))
```

### 四、高并发设计

#### 1. 请求合并（Request Merging / Prompt Batching）

将多个相似请求合并为一个批次发送给模型，降低推理成本。

```python
import asyncio
from collections import defaultdict

class RequestBatcher:
    """动态请求合并器"""

    def __init__(self, max_batch_size: int = 8, max_wait_ms: int = 50):
        self.max_batch_size = max_batch_size
        self.max_wait_ms = max_wait_ms
        self.queues = defaultdict(list)
        self._locks = defaultdict(asyncio.Lock)

    async def submit(self, prompt: str, model: str) -> str:
        future = asyncio.get_event_loop().create_future()
        async with self._locks[model]:
            self.queues[model].append((prompt, future))
            if len(self.queues[model]) >= self.max_batch_size:
                asyncio.create_task(self._process_batch(model))
            else:
                # 启动延迟批处理
                asyncio.create_task(self._delayed_process(model))
        return await future

    async def _delayed_process(self, model: str):
        await asyncio.sleep(self.max_wait_ms / 1000)
        async with self._locks[model]:
            if self.queues[model]:
                asyncio.create_task(self._process_batch(model))

    async def _process_batch(self, model: str):
        batch = self.queues[model][:self.max_batch_size]
        self.queues[model] = self.queues[model][self.max_batch_size:]
        # 批量调用模型
        prompts = [item[0] for item in batch]
        results = await batch_inference(model, prompts)
        for (_, future), result in zip(batch, results):
            future.set_result(result)
```

```python
import time
from collections import defaultdict

class SlidingWindowRateLimiter:
    """滑动窗口速率限制器"""

    def __init__(self, max_requests: int = 60, window_seconds: int = 60):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.requests = defaultdict(list)

    def is_allowed(self, key: str = "default") -> bool:
        now = time.time()
        window_start = now - self.window_seconds
        # 清理过期请求
        self.requests[key] = [
            t for t in self.requests[key] if t > window_start
        ]
        if len(self.requests[key]) >= self.max_requests:
            return False
        self.requests[key].append(now)
        return True

    def get_wait_time(self, key: str = "default") -> float:
        if self.is_allowed(key):
            return 0
        if not self.requests[key]:
            return 0
        # 窗口中最旧的请求过期时间
        oldest = min(self.requests[key])
        return max(0, oldest + self.window_seconds - time.time())
```

#### 4. 降级熔断（Circuit Breaker）

```python
from enum import Enum
import asyncio

class CircuitState(Enum):
    CLOSED = "closed"       # 正常运行
    OPEN = "open"           # 熔断
    HALF_OPEN = "half_open" # 半开（尝试恢复）

class CircuitBreaker:
    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 30):
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.last_failure_time = 0

    async def call(self, func, fallback_func=None, *args, **kwargs):
        if self.state == CircuitState.OPEN:
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
            else:
                if fallback_func:
                    return await fallback_func(*args, **kwargs)
                raise Exception("Circuit breaker is OPEN")

        try:
            result = await func(*args, **kwargs)
            if self.state == CircuitState.HALF_OPEN:
                self.state = CircuitState.CLOSED
                self.failure_count = 0
            return result
        except Exception as e:
            self.failure_count += 1
            self.last_failure_time = time.time()
            if self.failure_count >= self.failure_threshold:
                self.state = CircuitState.OPEN
            if fallback_func:
                return await fallback_func(*args, **kwargs)
            raise e
```

```python
from typing import AsyncIterator

class DataPipeline:
    def __init__(self):
        self.steps = []

    def add_step(self, name: str, processor):
        self.steps.append((name, processor))
```

```python
            "claude-3.5": ModelConfig(
                name="claude-3.5-sonnet", endpoint="https://api.anthropic.com/v1",
                cost_per_1k_tokens=0.02, max_tokens=8192,
                avg_latency_ms=1800, capabilities=["chat", "coding", "complex"]
            ),
            "llama3.1-70b": ModelConfig(
                name="llama3.1-70b", endpoint="http://vllm:8000/v1",
                cost_per_1k_tokens=0.005, max_tokens=8192,
                avg_latency_ms=2500, capabilities=["chat", "coding"]
            ),
            "qwen2.5-7b": ModelConfig(
                name="qwen2.5-7b", endpoint="http://vllm:8001/v1",
                cost_per_1k_tokens=0.001, max_tokens=4096,
                avg_latency_ms=500, capabilities=["chat", "simple"]
            ),
        }

    def route(self, task: dict) -> ModelConfig:
        """根据任务特征路由到最合适的模型"""
        complexity = task.get("complexity", "simple")
        latency_sla = task.get("latency_sla_ms", float("inf"))
        requires_vision = task.get("requires_vision", False)
        requires_coding = task.get("requires_coding", False)
        max_cost = task.get("max_cost", float("inf"))

        # 按成本排序，优先选低成本模型
        candidates = sorted(self.models.values(), key=lambda m: m.cost_per_1k_tokens)

        for model in candidates:
            if model.avg_latency_ms > latency_sla:
                continue
            if requires_vision and "vision" not in model.capabilities:
                continue
            if requires_coding and "coding" not in model.capabilities:
                continue
            if complexity == "complex" and "complex" not in model.capabilities:
                continue
            if model.cost_per_1k_tokens > max_cost:
                continue
            return model

        # Fallback到最便宜的模型
        return self.models["qwen2.5-7b"]

    def fallback(self, primary_model: str, error: Exception) -> ModelConfig:
        """主模型失败时切换到备用模型"""
        print(f"Model {primary_model} failed: {error}. Falling back...")
        # 按成本从低到高尝试
        for name, model in sorted(self.models.items(),
                                   key=lambda x: x[1].cost_per_1k_tokens):
            if name != primary_model:
                return model
        return self.models["qwen2.5-7b"]

    def a_b_test(self, task: dict, traffic_ratio: float = 0.1) -> tuple:
        """A/B测试：以指定比例路由到实验模型"""
        if random.random() < traffic_ratio:
            return "experimental", self.models["llama3.1-70b"]
        return "control", self.models["gpt-4o"]
```

#### 模型级联（Model Cascading）

```python
class ModelCascade:
    """模型级联：先用小模型，复杂度高时升级到大模型"""

    def __init__(self):
        self.small_model = "qwen2.5-1.5b"   # 成本低、速度快
        self.medium_model = "qwen2.5-7b"     # 适中
        self.large_model = "llama3.1-70b"    # 能力强、成本高

    async def infer_with_cascade(self, prompt: str) -> str:
        result = await self._call_model(self.small_model, prompt)
        if self._is_confident(result):
            return result
        # 小模型不够自信，升级到中模型
        result = await self._call_model(self.medium_model, prompt)
        if self._is_confident(result):
            return result
        # 最后使用大模型
        return await self._call_model(self.large_model, prompt)

    def _is_confident(self, result: str) -> bool:
        # 检查模型是否"不确定"
        uncertainty_markers = ["不确定", "不清楚", "我不知道", "I'm not sure", "cannot"]
        return not any(marker in result.lower() for marker in uncertainty_markers)
```
